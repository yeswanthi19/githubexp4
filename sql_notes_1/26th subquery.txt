-- Identify owner types where the total selling price of cars exceeds 1,000,000.
use kju;
select*from cars;
select owner,sum(selling_price) as total_price from cars group by owner having 
total_price > 1000000;

-- Find out brands that have more than one car listed in the dataset.

select brand,count(*) as total_car from cars group by brand having total_car>1;

-- get the minimum kilometers driven for cars in each fuel category.
select fuel,min(km_driven) as min_km from cars group by fuel order by min_km;
