-- Retrieve employee names along with their department names.
select emp_name,dept_name from emp t1 join dept t2 on
t1.dept_id=t2.dept_id;

-- Display all employees including those without a department.
select emp_name,dept_name from emp t1 left join dept t2 on
t1.dept_id=t2.dept_id;

-- Find employees not assigned to any department.
select emp_name,dept_name from emp t1 left join dept t2
on t1.dept_id=t2.dept_id where t2.dept_id is null;

-- Find departments with no employees.
select dept_name,emp_name from emp t1 right join dept t2
on t1.dept_id=t2.dept_id where t1.emp_name is null;

-- Display employee names with department locations.
select emp_name,dept_location from emp t1 join dept t2
on t1.dept_id=t2.dept_id;
-- Count number of employees in each department.
select dept_name ,count(emp_name) as total_emp
from emp t1 right join dept t2 on t1.dept_id=t2.dept_id
group by t2.dept_name;

-- List employees working in IT department.

select emp_name ,dept_name from emp t1 inner join dept t2 
on t1.dept_id=t2.dept_id 
where dept_name='it';
-- Display all combinations of employees and departments.

select* from emp t1 cross join dept t2;

Retrieve employees 
with salary greater than 60000 along with department names.
select salary,dept_name from emp t1 join dept t2
on t1.dept_id=t2.dept_id having salary> 60000;
